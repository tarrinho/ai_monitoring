   kubectl -n {{ .Release.Namespace }} port-forward svc/{{ include "ai-monitoring.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
   open http://localhost:{{ .Values.service.port }}/  (log in with MONITOR_ADMIN_USER / MONITOR_ADMIN_PASSWORD)

2. Prometheus metrics:
   http://{{ include "ai-monitoring.fullname" . }}.{{ .Release.Namespace }}.svc:{{ .Values.service.port }}/metrics
   {{- if .Values.serviceMonitor.enabled }}
   A ServiceMonitor was created — Prometheus Operator will scrape it (bearer: MONITOR_METRICS_TOKEN).
   {{- else }}
   Enable scraping with --set serviceMonitor.enabled=true (needs the Prometheus Operator CRD).
   {{- end }}
